type Collection a = [a]
type ItemsPerPage = Int

(//) :: Int -&gt; Int -&gt; Float
a // b = fromIntegral a / fromIntegral b

itemCount :: Collection a -&gt; Int
itemCount = length

pageCount :: Collection a -&gt; ItemsPerPage -&gt; Int
pageCount xs 0 = length xs
pageCount xs n = ceiling (length xs // n)

pageItemCount :: Collection a -&gt; ItemsPerPage -&gt; Int -&gt; Maybe Int
pageItemCount xs n page =
  let itemCount = length (drop (n * page) xs)
      res c
        | c &gt; n = Just n
        | c == 0 = Nothing
        | otherwise = Just c
  in if page &gt;= 0 then res itemCount else Nothing

pageIndex :: Show a =&gt; Collection a -&gt; ItemsPerPage -&gt; Int -&gt; Maybe Int
pageIndex xs n item =
  let pageNum = if n == 0 then item else floor (item // n)
  in
    if item &gt;= 0 &amp;&amp; item &lt; length xs then
      Just pageNum
    else
      Nothing```
